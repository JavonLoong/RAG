    "total_pages": 50         // 总页数
  },
  "annotations": [{
    "result": [
      {
        "id": "xxx",
        "type": "rectanglelabels",   // 标注框
        "value": {
          "rectanglelabels": ["Title"],  // 或 "Para"
          "y": 15.2                      // 纵坐标

        "id": "xxx",
        "type": "textarea",          // 文本内容
        "value": {
          "text": ["提取出的文字内容"]
        }
      }
    ]
  }]
}
